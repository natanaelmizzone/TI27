***Para mostra bancos de dados existentes***

SHOW DATABASES;






***Criar um banco de dados***

CREATE(criar) DATABASE db_escola(nome);


***Utiliza-se assim:***


CREATE DATABASE db_escola;







***Selecionar o database***

USE db_escola(nome do db);

***Utiliza-se assim:***

USE db_escola;



***Para criar tabela***

CREATE TABLE td_pessoas(NOME DA TABELA) 

Após inserir CREATE acima, clicar enter e inserir as informação da tabela, exemplo abaixo:

(nome VARCHAR(50), idade INT(3));





***Descrever a tabela****

Aplique o DESC

DESC td_pessoas;







***Inserir informações***

INSERT INTO td_pessoas (nome, idade)

Após inserir as informações acima, de enter e insira as informações abaixo:

VALUE ("Natanael Mizzone", 29);

Após inserir, de o enter.








Selecionar todas as colunas da tabela:

SELECT * FROM td_pessoas;

Assim facilita visualizar toda a tabela.

SELECT nome (para visualizar somente o nome) FROM td_pessoas;







Para filtrar maior que 30 em quantidade

SELECT * FROM estoque WHERE estoque > 30


Para atualizar os dados utilizar os comandos abaixo:

UPDATE (nome da tabela aqui, e de enter) 
SET nascimento (local da planilha a ser alterado)  = “2006-03-01” (dados a ser alterado)
WHERE nome(local direcionado para quem trocar) = “Nan Santos” (aqui o nome de quem, produto ou o que);

 



Para apagar uma linha:

DELETE FROM estoque WHERE nome = "detergente";





Comando que apaga toda a tabela:

DROPE  TABLE estoque(aqui inseri a tabela a ser apagada);
